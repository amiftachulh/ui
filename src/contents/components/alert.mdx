## Usage

```tsx
import {
  Alert,
  AlertContent,
  AlertDescription,
  AlertIcon,
  AlertTitle,
} from "@/components/ui/alert";
```

```tsx
<Alert variant="info">
  <AlertIcon>
    <LuInfo />
  </AlertIcon>
  <AlertContent>
    <AlertTitle>Info</AlertTitle>
    <AlertDescription>This is an info alert.</AlertDescription>
  </AlertContent>
</Alert>
```
