## Usage

```tsx
import {
  Field,
  FieldControl,
  FieldDescription,
  FieldError,
  FieldLabel,
} from "@/components/ui/field";
```

```tsx
<Field>
  <FieldLabel>Username</FieldLabel>
  <FieldControl>
    <input type="text" />
  </FieldControl>
  <FieldDescription>Enter your username</FieldDescription>
  <FieldError>Username is required</FieldError>
</Field>
```
