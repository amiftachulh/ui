## Usage

```tsx
import {
  Vaul,
  VaulClose,
  VaulContent,
  VaulDescription,
  VaulFooter,
  VaulHeader,
  VaulTitle,
  VaulTrigger,
} from "@/components/ui/vaul";
```

```tsx
<Vaul>
  <VaulTrigger>Open</VaulTrigger>
  <VaulContent>
    <VaulHeader>
      <VaulTitle>Are you absolutely sure?</VaulTitle>
      <VaulDescription>This action cannot be undone.</VaulDescription>
    </VaulHeader>
    <VaulFooter>
      <Button>Submit</Button>
      <VaulClose>
        <Button variant="outline">Cancel</Button>
      </VaulClose>
    </VaulFooter>
  </VaulContent>
</Vaul>
```
